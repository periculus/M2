              stmt_magic ← (lm_name, lm_arg)
          else:

              stmt_magic ← None

          balances ← count_delimiters(content)
          i ← i + 1

      """Track delimiter balancing and M2-specific completion rules."""

      def __init__(self):
          self.parens = 0
          self.braces = 0
          self.brackets = 0
          self.in_string = False
          self.in_triple_string = False
          self.in_block_comment = False
          self.string_char = None

      def update(self, line: str):
          """Update state based on line content."""
          i = 0
          while i < len(line):
              char = line[i]

              # Handle triple-slash strings
              if not self.in_string and not self.in_block_comment:

                          i += 3

                      self.in_block_comment = False
                      i += 2
                      continue
                  i += 1
                  continue

              if char == '(':

              elif char == ')':

              elif char == '}':

  def _has_hanging_operator(self, code: str) -> bool:
      """Check for operators at end that indicate continuation."""
      import re

      # Remove trailing whitespace
      code = code.rstrip()

      # Hanging binary operators
      hanging_patterns = [
          r'[+\-*/^%]\s*$',  # Arithmetic operators
          r'[=!<>]=?\s*$',   # Comparison operators  
          r'=\s*$',          # Assignment
          r'[|&]\s*$',       # Logical operators
          r'=>?\s*$',        # Arrow operators

      for pattern in hanging_patterns:
          if re.search(pattern, code):
              return True

      return False

  def _has_incomplete_control_structure(self, code: str) -> bool:
      """Check for incomplete if/for/while structures."""

          r'\belse\s*$',                      # hanging else
          r'\bdo\s*$',                        # hanging do
      ]

      for pattern in incomplete_patterns:
          if re.search(pattern, code, re.IGNORECASE | re.MULTILINE):
